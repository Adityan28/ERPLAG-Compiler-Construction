<<module array_parameter_demo>>
takes input[C:array[low..high] of integer];
start
	print(C);               
end
<<<driver program>>>
start
	declare m,n: integer;
	get_value(m);
	get_value(n);
	declare array_B: array[m..n] of integer;
	get_value(array_B);
	use module array_parameter_demo with parameters array_B;	
	print(array_B);
end
