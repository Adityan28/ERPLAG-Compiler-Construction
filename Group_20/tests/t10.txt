** Handles case and while statements codegen**
<<driver Program>>
start
	declare s:integer;
	declare t:real;
	declare u:boolean;
	declare a1,a2:integer;
	declare b1,b2:real;
	declare c1,c2:boolean;
	declare k: integer;

	a1 := 7;
	a2 := 12;
	b1 := 3.43;
	b2 := -1.6;
	s := 8;
	t := 15.75;
	
	for(k in 2..8)
	start
		switch(k)
		start
			case 2: u := a1<=a2;
				print(u);
				break;
			case 3: u := a1>a2;
				c1 = (s*a1+a2) != (a2*a1/s);
				print(u);
				print(c1)
				break;
			case 4: u := b1 >= a2;
				print(u);
				break;
			case 5: c2 := k+a2/a1 < k+a1/a2;
				print(c2);
				break;
			case 6: c1 := t-s==0;
				c2 := b1/2.35*b2-3.3*t < b1+b2;
				print(c1);
				print(u);
				break;
			default:u := FALSE;
				c1 := TRUE;
				c2 := TRUE;
				print(true);
		end
	end

	while(a1<a2 AND a1>0)
	start
		k := a1*a2-12;
		c1 := k>15 AND k<50;
		c2 := c2 AND c1;
		u := c2 OR c1 AND k>30;
		print(u);
		print(c1);
		print(c2);
		a1 := a1-1;
	end
end
