** For loop and arithmetic along with relational operations **
<<driver Program>>
start
	declare s:integer;
	declare t:real;
	declare a1,a2:integer;
	declare b1,b2:real;
	declare k: integer;
	declare u:array[4..10] of integer;
	declare v:array[4..10] of integer;
	declare w:array[3..7] of boolean;
	declare x:array[3..7] of real;
	
	a1 := 7;
	a2 := 12;
	b1 := 3.43;
	b2 := -1.6;
	s := 8;
	t := 15.75;
	
	for(k in 4..10)
	start
		v[k] := k+a2;
		print(v[k]);

		v[k] := a1*k-a2;
		print(v[k]);

		v[k] := (v[k]+k)*(v[k]-k);
		print(v[k])
	end

	u := v;
	print(v[5]);
	print(u[5]);
	print(v[8]);
	print(u[8]);

	for(k in 3..7)
	start
		w[k] := v[k+2]>=u[k+1];
		print(w[k]);

		w[k] := v[k+2]>u[k+1] AND v[5] != u[9];
		print(w[k]);

		w[k] := v[k+2]<=u[k+1] OR v[5] == u[9];
		print(w[k]);

		w[k] := v[k+2]<u[k+1];
		print(w[k]);
	end

	for(k in 3..7)
	start
		x[k] := b1+t;
		print(x[k]);

		x[k] := b1*t-b2;
		print(x[k]);

		x[k] := t/b1+b2;
		print(x[k]);
	end
end
